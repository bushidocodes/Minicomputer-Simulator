########################################################
# Test Vector                                          #
# writes two vectors of two starting at 33             #
# Watch addresses 33,34 in Debugger                    #
# Starts as 4,3,2,1                                    #
# VADD yields 2,2,2,1                                  #
# VSUB restores 4,3,2,1                                #
########################################################
LDX 1,0,16;          0 - Load IX1 with the base address of the stack frame
LDA 0,0,4;           1 - Local(1) = 4
STR 0,1,1;
LDA 0,0,3;           1 - Local(2) = 3
STR 0,1,2;
LDA 0,0,2;           1 - Local(3) = 2
STR 0,1,3;
LDA 0,0,1;           1 - Local(4) = 1
STR 0,1,4;
VSUB 2,1,1;          [Local(1), Local(2)] = [Local(1), Local(2)] - [Local(3), Local(4)]
VADD 2,1,1;          [Local(1), Local(2)] = [Local(1), Local(2)] + [Local(3), Local(4)]
TRAP 0;      end of program
